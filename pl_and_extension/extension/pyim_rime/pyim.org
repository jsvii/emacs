** 五笔输入法配置

#+BEGIN_SRC elisp
(require 'pyim)
;;(require 'pyim-basedict) ; 拼音词库设置，五笔用户 *不需要* 此行设置
;;(pyim-basedict-enable)   ; 拼音词库，五笔用户 *不需要* 此行设置
(setq default-input-method "pyim")
(pyim-wbdict-v98-enable) ; 启动输入法
#+END_SRC
